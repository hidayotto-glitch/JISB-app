💻 Aplikasi Kasir Sederhana (1 File)

Simpan sebagai: kasir.html

HTML
<!DOCTYPE html>
<html>
<head>
  <title>Kasir Sederhana</title>
  <style>
    body { font-family: Arial; padding: 20px; background:#f4f4f4;}
    h2 { text-align:center; }
    input, button { padding:8px; margin:5px; }
    table { width:100%; margin-top:15px; border-collapse: collapse;}
    th, td { border:1px solid #ccc; padding:8px; text-align:center;}
    button { cursor:pointer; }
    .total { font-size:18px; font-weight:bold; margin-top:10px;}
  </style>
</head>
<body>

<h2>APLIKASI KASIR SEDERHANA</h2>

<h3>Tambah Barang</h3>
<input type="text" id="nama" placeholder="Nama Barang">
<input type="number" id="harga" placeholder="Harga">
<input type="number" id="stok" placeholder="Stok">
<button onclick="tambahBarang()">Tambah</button>

<h3>Daftar Barang</h3>
<table>
  <thead>
    <tr>
      <th>Nama</th>
      <th>Harga</th>
      <th>Stok</th>
      <th>Beli</th>
    </tr>
  </thead>
  <tbody id="daftarBarang"></tbody>
</table>

<div class="total">Total: Rp <span id="total">0</span></div>
<button onclick="resetTotal()">Reset Transaksi</button>

<script>
let barang = JSON.parse(localStorage.getItem("barang")) || [];
let total = 0;

function simpanData(){
  localStorage.setItem("barang", JSON.stringify(barang));
}

function tampilkanBarang(){
  let tabel = document.getElementById("daftarBarang");
  tabel.innerHTML = "";
  barang.forEach((item, index) => {
    tabel.innerHTML += `
      <tr>
        <td>${item.nama}</td>
        <td>Rp ${item.harga}</td>
        <td>${item.stok}</td>
        <td><button onclick="beli(${index})">Beli</button></td>
      </tr>
    `;
  });
}

function tambahBarang(){
  let nama = document.getElementById("nama").value;
  let harga = parseInt(document.getElementById("harga").value);
  let stok = parseInt(document.getElementById("stok").value);

  if(nama && harga && stok){
    barang.push({nama, harga, stok});
    simpanData();
    tampilkanBarang();
    document.getElementById("nama").value="";
    document.getElementById("harga").value="";
    document.getElementById("stok").value="";
  }
}

function beli(index){
  if(barang[index].stok > 0){
    barang[index].stok -= 1;
    total += barang[index].harga;
    document.getElementById("total").innerText = total;
    simpanData();
    tampilkanBarang();
  } else {
    alert("Stok habis!");
  }
}

function resetTotal(){
  total = 0;
  document.getElementById("total").innerText = total;
}

tampilkanBarang();
</script>

</body>
</html>

💻 KODE APLIKASI KASIR + LAPORAN HARIAN

HTML
<!DOCTYPE html>
<html>
<head>
  <title>Kasir + Laporan Harian</title>
  <style>
    body { font-family: Arial; padding:20px; background:#f2f2f2;}

    input, button { padding:8px; margin:5px; }

    .box { background:white; padding:15px; margin-top:15px; border-radius:8px;}
    .total { font-weight:bold; font-size:18px; }
  </style>
</head>
<body>

<h2>APLIKASI KASIR + LAPORAN HARIAN</h2>

<div class="box">
<h3>Tambah Barang</h3>
<input type="text" id="nama" placeholder="Nama Barang">
<input type="number" id="modal" placeholder="Harga Modal">
<input type="number" id="harga" placeholder="Harga Jual">
<input type="number" id="stok" placeholder="Stok">
<button onclick="tambahBarang()">Tambah</button>
</div>

<div class="box">
<h3>Daftar Barang</h3>
<table>
<thead>
<tr>
<th>Nama</th>
<th>Harga</th>
<th>Stok</th>
<th>Beli</th>
</tr>
</thead>
<tbody id="daftarBarang"></tbody>
</table>
</div>

<div class="box">
<h3>Laporan Hari Ini</h3>
<p>Total Penjualan: Rp <span id="penjualan">0</span></p>
<p>Total Keuntungan: Rp <span id="keuntungan">0</span></p>
<button onclick="resetHarian()">Reset Laporan Hari Ini</button>
</div>

<script>
let barang = JSON.parse(localStorage.getItem("barang")) || [];
let laporan = JSON.parse(localStorage.getItem("laporan")) || {};
let today = new Date().toLocaleDateString();

if(!laporan[today]){
  laporan[today] = {penjualan:0, keuntungan:0};
}

function simpanData(){
  localStorage.setItem("barang", JSON.stringify(barang));
  localStorage.setItem("laporan", JSON.stringify(laporan));
}

function tampilkanBarang(){
  let tabel = document.getElementById("daftarBarang");
  tabel.innerHTML = "";
  barang.forEach((item,index)=>{
    tabel.innerHTML += `
      <tr>
        <td>${item.nama}</td>
        <td>Rp ${item.harga}</td>
        <td>${item.stok}</td>
        <td><button onclick="beli(${index})">Beli</button></td>
      </tr>
    `;
  });
}

function tambahBarang(){
  let nama = document.getElementById("nama").value;
  let modal = parseInt(document.getElementById("modal").value);
  let harga = parseInt(document.getElementById("harga").value);
  let stok = parseInt(document.getElementById("stok").value);

  if(nama && modal && harga && stok){
    barang.push({nama, modal, harga, stok});
    simpanData();
    tampilkanBarang();
    document.getElementById("nama").value="";
    document.getElementById("modal").value="";
    document.getElementById("harga").value="";
    document.getElementById("stok").value="";
  }
}

function beli(index){
  if(barang[index].stok > 0){
    barang[index].stok -= 1;

    laporan[today].penjualan += barang[index].harga;
    laporan[today].keuntungan += (barang[index].harga - barang[index].modal);

    updateLaporan();
    simpanData();
    tampilkanBarang();
  } else {
    alert("Stok habis!");
  }
}

function updateLaporan(){
  document.getElementById("penjualan").innerText = laporan[today].penjualan;
  document.getElementById("keuntungan").innerText = laporan[today].keuntungan;
}

function resetHarian(){
  laporan[today] = {penjualan:0, keuntungan:0};
  simpanData();
  updateLaporan();
}

updateLaporan();
tampilkanBarang();
</script>

</body>
</html>
